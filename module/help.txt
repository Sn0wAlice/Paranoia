# Paranoia Help
-help                 : show help message

# Creating partition
-c [size] [name]      : create partition (required args)

# Formatting partition
-f [name] [type]      : format partition (required args)
-fShow                : show available filesystems

# Zip
-z [name] [outName]   : zip file (required args)
-uz [name] [outName]  : unzip file (required args)

# Lock
-lock [path]          : lock file (required args)
-unlock [path]        : unlock file (required args)
